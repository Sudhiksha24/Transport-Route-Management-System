CREATE TABLE ROUTE_TB (
    ROUTEID VARCHAR2(30) PRIMARY KEY,
    ROUTECODE VARCHAR2(20) NOT NULL,
    ROUTENAME VARCHAR2(50) NOT NULL,
    STARTLOC VARCHAR2(50),
    ENDLOC VARCHAR2(50),
    DISTANCEKM NUMBER(8,2),
    REMARKS VARCHAR2(200)
);
CREATE SEQUENCE ROUTE_SEQ
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;
